You will need to download and buld Boost.Python (not available on Phonix as of September 2017) i.e. do something like the following:
cd ~/fastdir/ 
wget "http://dl.bintray.com/boostorg/release/1.65.0/source/boost_1_65_0.tar.gz"
tar -xvzf boost_1_65_0.tar.gz
cd boost_1_65_0
./bootstrap --with-libraries=python --prefix=~/fastdir/boost
./b2 install

# Change directory into wherever you have this repo /path/to/my/hed
make clean
module purge

# Python setup for HED
module load Python/2.7.13-foss-2016b
mkdir -p $FASTDIR/virtualenvs/hed
source $FASTDIR/virtualenvs/hed/bin/activate.csh # activate env
setenv PY_INC `python -c "from distutils import sysconfig; print sysconfig.get_python_inc()"` # Python include
setenv PY_INC `python -c "import numpy as np; print np.get_include()"`:$PY_INC # Numpy include
setenv PY_LIB `python -c "from distutils import sysconfig; print sysconfig.get_python_lib()"`
setenv BOOST_INCLUDE $FASTDIR/boost/include
setenv BOOST_LIB $FASTDIR/boost/lib
